    public static String serialize(Object obj) {
        StringBuilder sb = new StringBuilder();
        Class<?> clazz = obj.getClass();
        
        // è·å–æ‰€æœ‰å­—æ®µ
        Field[] fields = clazz.getDeclaredFields();
        for (Field field : fields) {

                  .append(field.get(obj))

                e.printStackTrace();
            }
        }
        return sb.toString();
    }
}
